export class RegisterComponent implements OnInit {

  // El form Group agrupa todo un formulario
  // El form control es un campo de un formulario
  // El form builder es una ayuda para crear formulario
  public formularioPrincipal: FormGroup
  public formularioSecundario: FormGroup
  @ViewChild('formPrinc') formPrinc: ElementRef;

  constructor(
    private fb : FormBuilder
  ) { }

  ngOnInit(): void {
    // Dos formas de hacer lo mismo
    // Para crear el formulario se puede crear de esta manera, en donde form Builder nos ayuda
    // Se recomienda para formularios grandes
    this.formularioPrincipal = this.fb.group({
      nombre: ['', [Validators.required, Validators.minLength(10)]],
      edad: ['', [Validators.required]],
      genero: ['', [Validators.required]]
    })
    // O tambien podemos hacerlo de esta manera en donde no utilizamos el builder sino que instanciamos para cada campo una instancia del form control
    // Para formularios chicos.
    this.formularioSecundario = new FormGroup({
      nombre: new FormControl(),
      edad: new FormControl(),
      genero: new FormControl()
    })

    // Con esta opcion estamos poniendo un escuchador
    this.formularioPrincipal.get('nombre')?.valueChanges.subscribe((value) => {
      console.log(value);
    })
  }
}
